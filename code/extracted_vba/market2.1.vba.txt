in file: /Users/dlev2617/Documents/Code/Agents/github-packages/READY-TO-UPLOAD/PUBLIC/paper-2003-04-a-model-for-optimizing-electronic-toll-collection-systems/code/original/market2.1.xls - OLE stream: '_VBA_PROJECT_CUR/VBA/Module1'

Sub Equilibrium()
Attribute Equilibrium.VB_Description = "Macro recorded 11/17/98 by Elva"
Attribute Equilibrium.VB_ProcData.VB_Invoke_Func = "T\n14"
'
' Equilibrium Macro
' Macro recorded 11/17/98 by Elva
'
' Keyboard Shortcut: Ctrl+Shift+T
'
    
    CoefT = Range("C33").Value
    CoefP = Range("C34").Value
        
    TotalLane = Range("D12").Value
    RampPrior = Range("D13").Value
    RampAfter = Range("D14").Value
    Speed = Range("D15").Value
    ACCDEC = 2 * 3600 * (RampPrior + RampAfter) / Speed
    ETCSurvival = Range("D16")
    AgencyBase = Worksheets("Results-Detail").Cells(31, 9).Value
    
    For i = 1 To 20
    
        'Initial Condition
            
       CoefZ = (Range("C32").Value) * (1 - i / 20)
       Duration = Cells(7 + i, 15).Value
        PeakArrival = Cells(7 + i, 16).Value
        ServiceETC = Cells(7 + i, 25).Value
        ServiceManual = Cells(7 + i, 19).Value
        PreETC = Cells(6 + i, 14).Value
        NPVStar = Worksheets("Results-Detail").Cells(5 + i, 21).Value
        For ETCLane = 1 To 11
 
            
        PriceDiff = 0
        DiscountStar = 0
        RelSplit = ETCLane / (TotalLane - ETCLane)
                
            Do
                ManualShare = (1 - ETCSurvival * PreETC) * (1 / (1 + RelSplit))
                ETCShare = 1 - ManualShare
                RHOETC = PeakArrival * ETCShare / (ETCLane * ServiceETC)
                RHOManual = PeakArrival * ManualShare / ((TotalLane - ETCLane) * ServiceManual)
                DelayETC = 900 * Duration * ((RHOETC - 1) + Sqr((RHOETC - 1) ^ 2 + 8 * RHOETC / _
                    (Duration * ServiceETC * ETCLane)))
                DelayManual = 900 * Duration * ((RHOManual - 1) + Sqr((RHOManual - 1) ^ 2 + _
                    8 * RHOManual / (Duration * ServiceManual * (TotalLane - ETCLane)))) + _
                    3600 / ServiceManual
                TimeDiff = (DelayETC - DelayManual - ACCDEC) / 60
                RelSplitNew = Exp(CoefZ + CoefT * TimeDiff + CoefP * PriceDiff)
                SplitDiff = Abs(RelSplitNew - RelSplit)
                RelSplit = (RelSplit + RelSplitNew) / 2
                
            Loop While SplitDiff > 0.00001
        
       Cells(7 + i, 24).Value = ETCLane
       Cells(7 + i, 14).Value = ETCShare
            Cells(7 + i, 9).Value = PriceDiff
            OverallNPV = Worksheets("Results-Detail").Cells(5 + i, 21).Value
            If (OverallNPV > NPVStar) Then
                NPVStar = OverallNPV
                DiscountStar = PriceDiff
                ETCShareStar = ETCShare
                SplitStar = ETCShareStar / (1 - ETCShareStar)
                SplitDiffStar = SplitDiff
                DelayETCStar = DelayETC
                DelayManualStar = DelayManual
                TimeDiffStar = TimeDiff
                ETCLaneStar = ETCLane
            End If
        Next
        Cells(7 + i, 11).Value = SplitStar
        Cells(7 + i, 12).Value = SplitDiffStar
        Cells(7 + i, 14).Value = ETCShareStar
        Cells(7 + i, 21).Value = DelayManualStar
        Cells(7 + i, 22).Value = ACCDEC
        Cells(7 + i, 27).Value = DelayETCStar
        Cells(7 + i, 9).Value = DiscountStar
        Cells(7 + i, 10).Value = TimeDiffStar
        Cells(7 + i, 13).Value = NPVStar
        Cells(7 + i, 11).NumberFormat = "0.00%"
        Cells(7 + i, 12).NumberFormat = "0.00%"
        Cells(7 + i, 14).NumberFormat = "0.00%"
        Cells(7 + i, 9).NumberFormat = "$#,##0.00"
        Cells(7 + i, 13).NumberFormat = "$#,##0"
        Worksheets("ETCLANE").Cells(7 + i, 10) = ETCShareStar
    Next
End Sub
